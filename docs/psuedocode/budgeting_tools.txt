define convert_currency function
    ask user for amount in US Dollars
    store as usd_amount

    ask user for target currency type
    store as target_type

    if target_type not in types
        display "Invalid currency type"
        stop

    converted_amount = usd_amount * types[target_type]

    display converted_amount and target_type

    when finished using converted_amount
        usd_amount = converted_amount / types[target_type]

    save usd_amount to file


define saving_goals function
    ask user for the name of the savings goal
    store as goal_name

    ask user for the amount they want to save
    store as saving_goal

    display goal_name and saving_goal with cash_type

    convert saving_goal back to USD by dividing by types[cash_type]
    save saving_goal in USD only


define budget_allocator function
    create empty Dictionary called categories

    ask user how many budget categories they want
    store as count

    repeat count times
        ask user for category name
        ask user for amount allocated to that category
        store in categories
    end repeat

    budget = sum of all category amounts

    display each category and amount in cash_type

    convert budget back to USD by dividing by types[cash_type]
    save budget in USD only


define interest_calculator function
    ask user for principal amount
    ask user for interest rate (decimal)
    ask user for time in years

    interest = principal * rate * time
    total = principal + interest

    display interest and total with cash_type

    convert total back to USD by dividing by types[cash_type]
    save total in USD only


define comp_interest function
    ask user for principal amount
    ask user for interest rate (decimal)
    ask user for number of years
    ask user for times compounded per year

    comp_interest = principal * (1 + rate/n)^(n * years)

    display comp_interest with cash_type

    convert comp_interest back to USD by dividing by types[cash_type]
    save comp_interest in USD only
